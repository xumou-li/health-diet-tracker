- ✅ `users` 表新增 `birthday DATE` 字段（用于精确计算年龄）
- ✅ `body_records` 表存储完整的身体快照（身高、体重、性别、年龄、活动水平、BMR、BMI、热量目标）
- ✅ `meal_records` 新增 `body_record_id BIGINT NOT NULL` 字段，关联身体状态
- ✅ **全程不使用外键**，符合简洁、实用、无耦合的项目规范
- ✅ 所有表含注释、合理索引、软删除、时间戳

```sql
-- 设置默认引擎和字符集
SET NAMES utf8mb4;
SET FOREIGN_KEY_CHECKS = 0;

-- 1. 用户表 users（新增 birthday 字段）
CREATE TABLE users (
    id BIGINT PRIMARY KEY AUTO_INCREMENT COMMENT '用户ID',
    phone VARCHAR(20) NOT NULL UNIQUE COMMENT '手机号（唯一）',
    email VARCHAR(100) DEFAULT NULL UNIQUE COMMENT '邮箱（可为空，唯一）',
    password_hash VARCHAR(255) NOT NULL COMMENT '密码哈希（bcrypt等）',
    gender TINYINT NOT NULL COMMENT '0=男, 1=女, 2=其他',
    birthday DATE NOT NULL COMMENT '出生日期（用于精确计算年龄）',
    height_cm INT NOT NULL COMMENT '当前身高（cm）',
    weight_kg DECIMAL(5,2) NOT NULL COMMENT '当前体重（kg）',
    activity_level TINYINT NOT NULL COMMENT '1=久坐, 2=轻度, 3=中度, 4=高强度',
    bmr INT NOT NULL COMMENT '当前基础代谢率（kcal/天）',
    daily_calorie_goal INT NOT NULL COMMENT '当前每日推荐热量（kcal）',
    bmi DECIMAL(4,2) NOT NULL COMMENT '当前BMI',
    is_deleted TINYINT(1) DEFAULT 0 COMMENT '软删除标志：0=正常，1=已删除',
    created_at DATETIME DEFAULT CURRENT_TIMESTAMP COMMENT '注册时间',
    updated_at DATETIME DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP COMMENT '最后更新身体档案时间'
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COMMENT='用户主表';

-- 2. 身体状态快照表 body_records（完整版）
CREATE TABLE body_records (
    id BIGINT PRIMARY KEY AUTO_INCREMENT,
    user_id BIGINT NOT NULL COMMENT '用户ID（无外键）',
    
    -- 身体参数快照（记录时的实际值）
    height_cm INT NOT NULL COMMENT '身高（cm）',
    weight_kg DECIMAL(5,2) NOT NULL COMMENT '体重（kg）',
    age INT NOT NULL COMMENT '记录时的实际年龄（由 birthday + recorded_at 计算得出）',
    gender TINYINT NOT NULL COMMENT '0=男,1=女,2=其他',
    activity_level TINYINT NOT NULL COMMENT '1=久坐,2=轻度,3=中度,4=高强度',
    
    -- 缓存计算结果
    bmi DECIMAL(4,2) NOT NULL COMMENT 'BMI = weight / (height/100)^2',
    bmr INT NOT NULL COMMENT '基础代谢率（kcal/天）',
    daily_calorie_goal INT NOT NULL COMMENT '当日推荐总热量（kcal）',
    
    recorded_at DATETIME DEFAULT CURRENT_TIMESTAMP COMMENT '快照时间',
    
    -- 索引：快速按用户+日期查找快照
    INDEX idx_user_date (user_id, DATE(recorded_at))
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COMMENT='用户身体状态快照（用于关联饮食记录）';

-- 3. 食物库表 foods
CREATE TABLE foods (
    id BIGINT PRIMARY KEY AUTO_INCREMENT,
    name VARCHAR(100) NOT NULL COMMENT '食物名称',
    category VARCHAR(30) NOT NULL COMMENT '分类：主食/蔬菜/肉类/水果/乳制品等',
    calorie_per_100g INT NOT NULL COMMENT '每100g热量（kcal）',
    protein_per_100g DECIMAL(5,2) NOT NULL COMMENT '蛋白质（g/100g）',
    carb_per_100g DECIMAL(5,2) NOT NULL COMMENT '碳水（g/100g）',
    fat_per_100g DECIMAL(5,2) NOT NULL COMMENT '脂肪（g/100g）',
    is_approved TINYINT(1) DEFAULT 1 COMMENT '是否审核通过：1=是，0=否',
    created_by BIGINT DEFAULT NULL COMMENT '创建者管理员ID',
    created_at DATETIME DEFAULT CURRENT_TIMESTAMP
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COMMENT='标准食物营养数据库';

-- 4. 用户常用食物收藏表 favorite_foods
CREATE TABLE favorite_foods (
    id BIGINT PRIMARY KEY AUTO_INCREMENT,
    user_id BIGINT NOT NULL COMMENT '用户ID',
    food_id BIGINT NOT NULL COMMENT '食物ID',
    created_at DATETIME DEFAULT CURRENT_TIMESTAMP,
    INDEX idx_user_food (user_id, food_id)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COMMENT='用户收藏的常用食物';

-- 5. 饮食记录表 meal_records（关键：关联 body_record_id）
CREATE TABLE meal_records (
    id BIGINT PRIMARY KEY AUTO_INCREMENT,
    user_id BIGINT NOT NULL COMMENT '用户ID',
    food_id BIGINT NOT NULL COMMENT '食物ID',
    body_record_id BIGINT NOT NULL COMMENT '对应的身体快照ID（核心关联字段）',
    
    date DATE NOT NULL COMMENT '用餐日期（如 2026-01-24）',
    meal_type TINYINT NOT NULL COMMENT '1=早餐, 2=午餐, 3=晚餐, 4=加餐',
    weight_g DECIMAL(7,2) NOT NULL COMMENT '实际摄入重量（克）',
    
    -- 营养值快照（基于 foods 和 weight_g 计算后存储）
    calorie INT NOT NULL COMMENT '实际热量（kcal）',
    protein DECIMAL(6,2) NOT NULL COMMENT '实际蛋白（g）',
    carb DECIMAL(6,2) NOT NULL COMMENT '实际碳水（g）',
    fat DECIMAL(6,2) NOT NULL COMMENT '实际脂肪（g）',
    
    created_at DATETIME DEFAULT CURRENT_TIMESTAMP,
    
    -- 索引优化查询性能
    INDEX idx_user_date (user_id, date),
    INDEX idx_body_record (body_record_id)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COMMENT='用户饮食记录（绑定身体状态快照）';

-- 6. AI建议调用记录表 ai_suggestions
CREATE TABLE ai_suggestions (
    id BIGINT PRIMARY KEY AUTO_INCREMENT,
    user_id BIGINT NOT NULL COMMENT '用户ID',
    content TEXT NOT NULL COMMENT 'AI生成的建议文本',
    triggered_by VARCHAR(50) NOT NULL COMMENT '触发条件标识',
    created_at DATETIME DEFAULT CURRENT_TIMESTAMP,
    INDEX idx_user_date (user_id, DATE(created_at))
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COMMENT='AI营养建议调用记录';

-- 7. 系统配置表 system_configs
CREATE TABLE system_configs (

    ai_enabled TINYINT(1) DEFAULT 1 COMMENT '全局AI功能开关',
    ai_prompt_template TEXT COMMENT 'AI提示词模板',
    updated_at DATETIME DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COMMENT='系统全局配置';

-- 恢复外键检查（虽未使用，但保持安全）
SET FOREIGN_KEY_CHECKS = 1;
```

### 💡 使用场景示例

> **问题**：用户想查看“2026-01-10 这一天我吃了多少热量？当时我的目标是多少？”  
> **解决方案**：
> ```sql
> SELECT 
> SUM(m.calorie) AS total_intake,
> br.daily_calorie_goal AS target
> FROM meal_records m
> JOIN body_records br ON m.body_record_id = br.id
> WHERE m.user_id = 123 AND m.date = '2026-01-10'
> GROUP BY br.daily_calorie_goal;
> ```
